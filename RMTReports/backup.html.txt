<script>
    document.addEventListener('DOMContentLoaded', function () {
        // Fetch the RMTR number from the server
        fetch('/generate-rmtr-number/')
            .then(response => {
                if (!response.ok) {
                    throw new Error('Failed to fetch RMTR number');
                }
                return response.json();
            })
            .then(data => {
                const rmtrNoInput = document.getElementById('rmtr-no');
                if (rmtrNoInput && data.rmtr_number) {
                    rmtrNoInput.value = data.rmtr_number;
                }
            })
            .catch(error => console.error('Error fetching RMTR number:', error));

        // Fetch supplier options from the backend
        fetch('/api/suppliers')
            .then(response => {
                if (!response.ok) {
                    throw new Error('Failed to fetch suppliers');
                }
                return response.json();
            })
            .then(suppliers => {
                const supplierSelect = document.getElementById('supplier');
                if (supplierSelect) {
                    suppliers.forEach(supplier => {
                        const option = document.createElement('option');
                        option.value = supplier.id;
                        option.textContent = supplier.name;
                        supplierSelect.appendChild(option);
                    });
                }
            })
            .catch(error => console.error('Error fetching suppliers:', error));

        // HOD Mapping based on plant type
        const plantHodMap = {
            'refinery': 'Francis Muthangya',
            'tissue': 'Stephen Ogot',
            'baking': 'Mary Melvin',
            'soap': 'Rushabh SHAH',
            'detergent': 'Felix Musyoki'
        };

        window.updateHOD = function () {
            const plant = document.getElementById('plant')?.value;
            const hodInput = document.getElementById('hod');
            if (hodInput) {
                hodInput.value = plantHodMap[plant] || ''; // Set to empty string if not found
            }
        };

        // Material and Test Options Handling
        const materialSubCategories = {
            food: {
                "Edible Oils": ['Moisture', 'Smoke point', 'Flavour profile', 'Nutritional value', 'Heavy metals', 'Colour', 'FFA', 'Density', 'Final cooked product'],
                "Flavours": ['Texture', 'Flavour taste within Final cooked product'],
                "Fragrances": ['Smells', 'Miscibility', 'Fragrance longevity', 'Dry-down']
            },
            "non-food": {
                "Chemical/Ingredient": ['Oils', 'Flavours', 'Fragrance', 'Texture', 'Moisture', 'Final FG performance'],
                "Oils": ['Moisture', 'Smoke point', 'Flavour profile', 'Nutritional value', 'Heavy metals', 'Colour', 'FFA', 'Density', 'Final cooked product'],
                "Colour": ['Visual', 'Miscibility', 'Stability', 'Wash test']
            },
            "packaging": {
                "Cartons": ['Fitment', 'Dimensions', 'GSM', 'Sealing', 'Stacking'],
                "Tub/Buckets": ['Weight tests', 'Visible damages', 'Visible impurities', 'Flexibility', 'Colour matching', 'Lid fitting'],
                "PET Jars": ['Weight tests', 'Clarity', 'Cap fitting', 'Handle fitting', 'Visible damages']
            }
        };

        const materialTypeSelect = document.getElementById('material-type');
        const subCategorySelect = document.getElementById('sub-category');
        const subCategorySection = document.getElementById('sub-category-section');
        const testsModal = document.getElementById('testsModal');
        const testsContainer = document.getElementById('tests-container');
        const testsSection = document.getElementById('tests-section');
        const selectedTestsList = document.getElementById('selected-tests-list');
        const submitModalButton = document.getElementById('submit-modal');
        let selectedTests = [];

        // Event listener for material type dropdown
        materialTypeSelect.addEventListener('change', function () {
            const selectedMaterialType = materialTypeSelect.value;
            if (selectedMaterialType) {
                populateSubCategories(selectedMaterialType);
                subCategorySection.classList.remove('hidden');
                testsSection.classList.add('hidden');
            } else {
                subCategorySection.classList.add('hidden');
                testsSection.classList.add('hidden');
            }
        });

        // Event listener for sub-category dropdown
        subCategorySelect.addEventListener('change', function () {
            const selectedMaterialType = materialTypeSelect.value;
            const selectedSubCategory = subCategorySelect.value;
            if (selectedSubCategory) {
                openModal(selectedMaterialType, selectedSubCategory);
            }
        });

        // Populate sub-categories based on selected material type
        function populateSubCategories(materialType) {
            subCategorySelect.innerHTML = ''; // Clear previous options
            subCategorySelect.appendChild(new Option('--Select Sub-Category--', ''));
            const subCategories = Object.keys(materialSubCategories[materialType] || {});
            subCategories.forEach(subCategory => {
                const option = document.createElement('option');
                option.value = subCategory;
                option.textContent = subCategory;
                subCategorySelect.appendChild(option);
            });
        }

        // Open modal with tests for the selected sub-category
        function openModal(materialType, subCategory) {
            testsContainer.innerHTML = '';  // Clear previous tests
            selectedTests = [];
            const tests = materialSubCategories[materialType]?.[subCategory] || [];

            tests.forEach(test => {
                const label = document.createElement('label');
                const checkbox = document.createElement('input');
                checkbox.type = 'checkbox';
                checkbox.value = test;

                label.appendChild(checkbox);
                label.appendChild(document.createTextNode(test));
                testsContainer.appendChild(label);
            });

            // Show modal
            testsModal.style.display = 'block';
        }

        // Close modal and save selected tests
        submitModalButton.addEventListener('click', saveAndCloseModal);
        document.querySelector('.close-button').addEventListener('click', saveAndCloseModal);

        function saveAndCloseModal() {
            // Gather selected tests
            const checkboxes = testsContainer.querySelectorAll('input[type="checkbox"]:checked');
            selectedTests = Array.from(checkboxes).map(checkbox => checkbox.value);

            // Display selected tests
            selectedTestsList.innerHTML = selectedTests.length > 0 ? selectedTests.join(', ') : 'No tests selected';
            testsSection.classList.remove('hidden');

            // Close modal
            testsModal.style.display = 'none';
        }

        // Close modal if clicking outside of it
        window.onclick = function(event) {
            if (event.target === testsModal) {
                saveAndCloseModal();
            }
        };

        // Toggle supplier form visibility
        window.toggleAddSupplierForm = function () {
            const addSupplierForm = document.getElementById('add-supplier-form');
            if (addSupplierForm) {
                addSupplierForm.style.display = addSupplierForm.style.display === 'none' ? 'block' : 'none';
            }
        };

        // Add supplier
        window.addSupplier = async () => {
            const supplierName = document.getElementById('new-supplier-name')?.value;
            if (!supplierName) {
                alert('Please enter a supplier name.');
                return;
            }

            const csrftoken = document.querySelector('[name=csrfmiddlewaretoken]').value;
            const supplierData = { name: supplierName };

            const response = await fetch('/manage-suppliers/', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                    'X-CSRFToken': csrftoken,
                },
                body: JSON.stringify(supplierData)
            });

            if (response.ok) {
                const data = await response.json();
                console.log('Supplier added:', data);
                const supplierSelect = document.getElementById('supplier');
                if (supplierSelect) {
                    const option = document.createElement('option');
                    option.value = data.id;
                    option.textContent = data.name;
                    supplierSelect.appendChild(option);
                }
                document.getElementById('new-supplier-name').value = '';
                toggleAddSupplierForm();
            } else {
                console.error('Error adding supplier:', response.statusText);
            }
        };

                // Utility function to handle form submission
        async function handleFormSubmission(buttonName, event) {
            event.preventDefault(); // Prevent default form submission

            const form = document.getElementById('test-form');
            const formData = new FormData(form);
            const csrftoken = document.querySelector('[name=csrfmiddlewaretoken]').value;

            // Check if CSRF token exists
            if (!csrftoken) {
                console.error('CSRF token missing');
                return;
            }

            const messageDiv = document.getElementById('form-message');
            const errorDiv = document.getElementById('form-error');
            const submitButton = event.target;

            // Hide any previous messages
            messageDiv.style.display = 'none';
            errorDiv.style.display = 'none';

            // Disable the button and show a loading state
            submitButton.disabled = true;
            submitButton.textContent = 'Processing...';

            try {
                // Send the form data via fetch
                const response = await fetch(form.action, {
                    method: 'POST',
                    headers: {
                        'X-CSRFToken': csrftoken,
                    },
                    body: formData
                });

                // Show appropriate messages based on response
                if (response.ok) {
                    const successMessage = buttonName === 'submit'
                        ? 'Form submitted successfully!'
                        : 'Form saved successfully!';
                    messageDiv.textContent = successMessage;
                    messageDiv.style.display = 'block'; // Show success message

                    // Redirect to dashboard after a successful submission or save
                    window.location.href = '/dashboard/'; // Or any other redirect URL
                } else {
                    const errorMessage = await response.json();
                    errorDiv.textContent = 'Error: ' + (errorMessage.message || 'An unknown error occurred');
                    errorDiv.style.display = 'block'; // Show error message
                }
            } catch (error) {
                console.error('Error during fetch:', error);
                errorDiv.textContent = 'An error occurred while processing your request.';
                errorDiv.style.display = 'block'; // Show generic error message
            } finally {
                // Re-enable the button and reset text after processing
                submitButton.disabled = false;
                submitButton.textContent = buttonName === 'submit' ? 'Submit' : 'Save';
            }
        }

        // Attach event listener to submit button
        document.querySelector('button[name="submit"]').addEventListener('click', function(event) {
            handleFormSubmission('submit', event);
        });

        // Attach event listener to save button
        document.querySelector('button[name="save"]').addEventListener('click', function(event) {
            handleFormSubmission('save', event);
        });

        
    });
</script>
